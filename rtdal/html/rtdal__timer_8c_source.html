50 int timer_setup (rtdal_timer_t *info)
51 {
52  long int ns;
53  __time_t sec;
54  int fd;
55 
56  if (info->period<=0) {
57  aerror_msg("Invalid period %d\n", (int) info->period);
58  return -1;
59  }
60 
61  /* Create the timer */
62  fd = timerfd_create (CLOCK_REALTIME, 0);
63  if (fd == -1) {
64  perror("timerfd_create");
65  return -1;
66  }
67 
68  info->wakeups_missed = 0;
69  info->timer_fd = fd;
70 
71  /* Make the timer periodic */
72  sec = info->period/1000000000;
73  ns = (info->period - (sec * 1000000));
74  info->itval.it_interval.tv_sec = sec;
75  info->itval.it_interval.tv_nsec = ns;
76  info->itval.it_value.tv_sec = sec;
77  info->itval.it_value.tv_nsec = ns;
78 
79  return 0;
80 }

93 inline int timer_wait_period (rtdal_timer_t *info)
94 {
95  unsigned long long missed;
96  int ret;
97 
98  /* Wait for the next timer event. If we have missed any the
99  number is written to "missed" */
100  ret = read (info->timer_fd, &missed, sizeof (missed));
101  if (ret == -1)
102  {
103  perror ("read timer");
104  return ret;
105  }
106 
107  /* "missed" should always be >= 1, but just to be sure, check it is not 0 anyway */
108  if (missed > 0)
109  info->wakeups_missed += (missed - 1);
110 
111  return 0;
112 }

142 void* timerfd_timer_run_thread(rtdal_timer_t* obj) {
143  int s;
144 
145  assert(obj->period_function);
146 
147  if (timer_setup(obj)) {
148  aerror("Setting up timer");
149  goto timer_destroy;
150  }
151 
152  if (obj->wait_futex) {
153  futex_wait(obj->wait_futex);
154  }
155  if (timer_start(obj)) {
156  aerror("timer_start");
157  goto timer_destroy;
158  }
159 
160  obj->stop = 0;
161  while(!obj->stop) {
162  if (timer_wait_period(obj)) {
163  aerror("waiting for timer");
164  goto timer_destroy;
165  }
166  obj->period_function(obj->arg, NULL);
167  }
168 
169  s = 0;
170 
171 timer_destroy:
172  timer_close(obj);
173  pthread_exit(&s);
174  return NULL;
175 }

177 inline static void timespec_add_us(struct timespec *t, long int us) {
178  t->tv_nsec += us;
179  if (t->tv_nsec >= 1000000000) {
180  t->tv_nsec = t->tv_nsec - 1000000000;

185  t->tv_nsec -= us;
186  if (t->tv_nsec < 0) {
187  t->tv_nsec = t->tv_nsec + 1000000000;
188  t->tv_sec -= 1;

192 void* nanoclock_timer_run_thread(rtdal_timer_t* obj) {
193  int s;
194  int n;
195  assert(obj->period_function);
196 
197  obj->stop = 0;
198  if (obj->wait_futex) {
199  futex_wait(obj->wait_futex);
200  obj->next.tv_sec+=TIMER_FUTEX_GUARD_SEC;
201  obj->next.tv_nsec=0;
202  } else {
203  clock_gettime(CLOCK_REALTIME, &obj->next);
204  }
205  n=0;
206  while(!obj->stop) {
207  timespec_add_us(&obj->next, obj->period);
208  clock_nanosleep(CLOCK_REALTIME, TIMER_ABSTIME,
209  &obj->next, NULL);
210 
211  n++;
212  if (n>=obj->multiple) {
213  obj->period_function(obj->arg, &obj->next);
214  n=0;
215  }
216  }
217 
218  s = 0;
219  pthread_exit(&s);
220  return NULL;
221 }
