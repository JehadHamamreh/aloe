55 void rx_thread(uhd::usrp::multi_usrp::sptr usrp, const std::string &rx_cpu,
56  const std::string &rx_otw, struct dac_cfg *cfg, void (*sync)(void), long int *tslen_ptr){
57 
58  double rx_rate=0;
59 
60  //create a receive streamer
61  uhd::stream_args_t stream_args(rx_cpu, rx_otw);
62  stream_args.channels.push_back(0); //single-channel
63  uhd::rx_streamer::sptr rx_stream = usrp->get_rx_stream(stream_args);
64 
65  //setup variables and allocate buffer
66  uhd::rx_metadata_t md;
67 
68  std::vector<void *> buffs;
69  buffs.push_back(cfg->dacinbuff[0]); //only 1 channel is used
70  bool had_an_overflow = false;
71  uhd::time_spec_t last_time;
72  const double rate = usrp->get_rx_rate();
73 
74  uhd::stream_cmd_t cmd(uhd::stream_cmd_t::STREAM_MODE_START_CONTINUOUS);
75  cmd.time_spec = usrp->get_time_now() + uhd::time_spec_t(0.05);
76  cmd.stream_now = (buffs.size() == 1);
77  usrp->issue_stream_cmd(cmd);
78  while (not boost::this_thread::interruption_requested()){
79  /* Poll if sampling frequency has changed */
80  if (cfg->inputFreq!=rx_rate) {
81  usrp->set_rx_rate(cfg->inputFreq);
82  rx_rate=cfg->inputFreq;
83  }
84 
85  double x=(double) cfg->NsamplesOut/rx_rate;
86  *tslen_ptr=(long int) 1000000*x;
87 
88  num_rx_samps += rx_stream->recv(buffs, cfg->NsamplesIn, md);
89 
90  sync();
91 
92  //handle the error codes
93  switch(md.error_code){
94  case uhd::rx_metadata_t::ERROR_CODE_NONE:
95  if (had_an_overflow){
96  had_an_overflow = false;
97  num_dropped_samps += (int) (md.time_spec - last_time).get_real_secs()*rate;
98  }
99  break;
100 
101  case uhd::rx_metadata_t::ERROR_CODE_OVERFLOW:
102  had_an_overflow = true;
103  last_time = md.time_spec;
104  num_overflows++;
105  break;
106 
107  default:
108  std::cerr << "Error code: " << md.error_code << std::endl;
109  std::cerr << "Unexpected error on recv, continuing..." << std::endl;
110  break;
111  }
112 
113  }
114  usrp->issue_stream_cmd(uhd::stream_cmd_t::STREAM_MODE_STOP_CONTINUOUS);
115 }

189  uhd::async_metadata_t async_md;
190 
191  while (not boost::this_thread::interruption_requested()){
192 
193  if (not usrp->get_device()->recv_async_msg(async_md)) continue;
194 
195  //handle the error codes
196  switch(async_md.event_code){
197  case uhd::async_metadata_t::EVENT_CODE_BURST_ACK:
198  return;
199 
200  case uhd::async_metadata_t::EVENT_CODE_UNDERFLOW:
201  case uhd::async_metadata_t::EVENT_CODE_UNDERFLOW_IN_PACKET:
202  num_underflows++;
203  break;
204 
205  case uhd::async_metadata_t::EVENT_CODE_SEQ_ERROR:
206  case uhd::async_metadata_t::EVENT_CODE_SEQ_ERROR_IN_BURST:
207  num_seq_errors++;
208  break;
209 
210  default:
211  std::cerr << "Event code: " << async_md.event_code << std::endl;

239 int uhd_init(struct dac_cfg *cfg, long int *timeSlotLength, void (*sync)(void)){
240 
241  std::string rx_otw, tx_otw;
242  std::string rx_cpu, tx_cpu;
243  rx_otw="sc16";
244  tx_otw="sc16";
245 
246  if (cfg->sampleType) {
247  rx_cpu="sc16";
248  tx_cpu="sc16";
249  } else {
250  rx_cpu="fc32";
251  tx_cpu="fc32";
252  }
253 
254  //spawn the receive test thread
255  if (main_cfg.chain_is_tx==0) {
257  for(size_t chan = 0; chan < usrp->get_rx_num_channels(); chan++) {
258  double freq = cfg->inputRFFreq;
259  if (freq>0) {
260  usrp->set_rx_freq(freq, chan);
261  }
262 
263  //set the rf gain
264  double gain = cfg->rx_gain;
265  if (gain>0) {
266  usrp->set_rx_gain(gain, chan);
267  }
268 
269  //set the IF filter bandwidth
270  double bw = cfg->rx_bw;
271  if (bw>0) {
272  usrp->set_rx_bandwidth(bw, chan);
273  }
274  }
275 
276  double x=(double) cfg->NsamplesOut/cfg->inputFreq;
277  *timeSlotLength=(int) 1000000*x;
278  thread_group.create_thread(boost::bind(&rx_thread, usrp, rx_cpu, rx_otw,cfg,sync,timeSlotLength));
279  }
280 
281  //spawn the transmit test thread
282  if (main_cfg.chain_is_tx==1) {
283 
285  for(size_t chan = 0; chan < usrp->get_tx_num_channels(); chan++) {
286  double freq = cfg->outputRFFreq;
287  if (freq>0) {
288  usrp->set_tx_freq(freq, chan);
289  }
290 
291  //set the rf gain
292  double gain = cfg->tx_gain;
293  if (gain>0.0) {
294  usrp->set_tx_gain(gain, chan);
295  }
296 
297  //set the IF filter bandwidth
298  double bw = cfg->tx_bw;
299  if (bw>0.0) {
300  usrp->set_tx_bandwidth(bw, chan);
301  }
302  printf("\nTX Chain: Freq=%g MHz\tBW=%g MHz\tGain=%g dB\n",freq/1000000,bw/1000000,gain);
303  }
304 
305  double x=(double) cfg->NsamplesOut/cfg->outputFreq;
306  *timeSlotLength=(int) 1000000*x;
307  thread_group.create_thread(boost::bind(&tx_thread, usrp, tx_cpu, tx_otw,cfg,sync,timeSlotLength));
308  }
309  return 1;
310 }
