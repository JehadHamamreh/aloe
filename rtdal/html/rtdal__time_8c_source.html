127 void rtdal_time_interval(time_t * tdata) {
128  assert(tdata);
129 
130  tdata[0].tv_sec = tdata[2].tv_sec - tdata[1].tv_sec;
131  tdata[0].tv_usec = tdata[2].tv_usec - tdata[1].tv_usec;
