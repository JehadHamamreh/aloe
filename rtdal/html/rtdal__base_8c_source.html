118 int rtdal_periodic_add(void (*callback)(void), int period) {
119  assert(context);
120  RTDAL_ASSERT_PARAM(callback);
121  RTDAL_ASSERT_PARAM(period>0);
122 
123  pthread_mutex_lock(&context->mutex);
124 
125  int i;
126 
127  for (i=0;i<MAX(rtdal_periodic);i++) {
128  if (!context->periodic[i].callback)
129  break;
130  }
131  if (i == MAX(rtdal_periodic)) {
132  RTDAL_SETERROR(RTDAL_ERROR_NOSPACE);
133  pthread_mutex_unlock(&context->mutex);
134  return -1;
135  }
136  context->periodic[i].counter = 0;
137  context->periodic[i].period = period;
138  context->periodic[i].callback = callback;
139  pthread_mutex_unlock(&context->mutex);
140  hdebug("i=%d, period=%d, callback=0x%x\n",i,period,callback);
141  return 0;
142 }

312 r_proc_t rtdal_process_new(struct rtdal_process_attr *attr, void *arg) {
313  hdebug("binary=%s, proc=%d, pos=%d, finish=0x%x, arg=0x%x\n",attr->binary_path,attr->pipeline_id,
314  attr->exec_position,attr->finish_callback,arg);
315  assert(context);
316  RTDAL_ASSERT_PARAM_P(attr);
317  RTDAL_ASSERT_PARAM_P(arg);
318 
319  pthread_mutex_lock(&context->mutex);
320  int i=0;
321  /* find empty space on process db */
322  for (i=0;i<MAX(rtdal_process);i++) {
323  if (!context->processes[i].pid)
324  break;
325  }
326 
327  if (i == MAX(rtdal_process)) {
328  RTDAL_SETERROR(RTDAL_ERROR_NOSPACE);
329  goto out;
330  }
331 
332  if (attr->process_group_id > MAX_PROCESS_GROUP_ID) {
333  RTDAL_SETERROR(RTDAL_ERROR_NOSPACE);
334  goto out;
335  }
336 
337  if (attr->process_group_id < 0) {
338  RTDAL_SETERROR(RTDAL_ERROR_INVAL);
339  goto out;
340  }
341 
342  pgroup_notified_failure[attr->process_group_id] = 0;
343  memset(&context->processes[i],0,sizeof(rtdal_process_t));
344  context->processes[i].pid=i+1;
345 
346  hdebug("i=%d, pid=%d\n",context->processes[i].pid);
347 
348  memcpy(&context->processes[i].attributes, attr,
349  sizeof(struct rtdal_process_attr));
350  context->processes[i].arg = arg;
351  context->processes[i].finish_code = FINISH_OK;
352 
353  if (pipeline_add(&context->pipelines[attr->pipeline_id],
354  &context->processes[i]) == -1) {
355  goto out;
356  }
357 
358  if (rtdal_process_launch(&context->processes[i])) {
359  pipeline_remove(&context->pipelines[attr->pipeline_id],
360  &context->processes[i]);
361  goto out;
362  }
363 
364  pthread_mutex_unlock(&context->mutex);
365  return (r_proc_t) &context->processes[i];
366 out:
367  pthread_mutex_unlock(&context->mutex);
368  return NULL;
369 }
