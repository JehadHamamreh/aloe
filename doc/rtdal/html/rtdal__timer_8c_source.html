45 int timer_setup (rtdal_timer_t *info)
46 {
47  long int ns;
48  __time_t sec;
49  int fd;
50 
51  if (info->period<=0) {
52  aerror_msg("Invalid period %d\n", (int) info->period);
53  return -1;
54  }
55 
56  /* Create the timer */
57  fd = timerfd_create (CLOCK_MONOTONIC, 0);
58  if (fd == -1) {
59  perror("timerfd_create");
60  return -1;
61  }
62 
63  info->wakeups_missed = 0;
64  info->timer_fd = fd;
65 
66  /* Make the timer periodic */
67  sec = info->period/1000000000;
68  ns = (info->period - (sec * 1000000));
69  info->itval.it_interval.tv_sec = sec;
70  info->itval.it_interval.tv_nsec = ns;
71  info->itval.it_value.tv_sec = sec;
72  info->itval.it_value.tv_nsec = ns;
73 
74  return 0;
75 }

88 inline int timer_wait_period (rtdal_timer_t *info)
89 {
90  unsigned long long missed;
91  int ret;
92 
93  /* Wait for the next timer event. If we have missed any the
94  number is written to "missed" */
95  ret = read (info->timer_fd, &missed, sizeof (missed));
96  if (ret == -1)
97  {
98  perror ("read timer");
99  return ret;
100  }
101 
102  /* "missed" should always be >= 1, but just to be sure, check it is not 0 anyway */
103  if (missed > 0)
104  info->wakeups_missed += (missed - 1);
105 
106  return 0;
107 }

109 void* timer_run_thread(void* x) {
110  rtdal_timer_t *obj = (rtdal_timer_t*) x;
111 
112  if (DEBUG_rtdal) {
113  rtdal_task_print_sched();
114  }
115 
116  switch(obj->mode) {
117  case NANOSLEEP:
118  return nanoclock_timer_run_thread(obj);
119  case TIMERFD:
120  return timerfd_timer_run_thread(obj);
121  default:
122  aerror_msg("Unknown timer mode %d\n", obj->mode);
123  return NULL;
124  }

127 void* timerfd_timer_run_thread(rtdal_timer_t* obj) {
128  int s;
129 
130  assert(obj->period_function);
131 
132  if (timer_setup(obj)) {
133  aerror("Setting up timer");
134  goto timer_destroy;
135  }
136 
137  if (obj->wait_futex) {
138  futex_wait(obj->wait_futex);
139  }
140  if (timer_start(obj)) {
141  aerror("timer_start");
142  goto timer_destroy;
143  }
144 
145  obj->stop = 0;
146  while(!obj->stop) {
147  if (timer_wait_period(obj)) {
148  aerror("waiting for timer");
149  goto timer_destroy;
150  }
151  obj->period_function(obj->arg, NULL);
152  }
153 
154  s = 0;
155 
156 timer_destroy:
157  timer_close(obj);
158  pthread_exit(&s);
159  return NULL;
160 }

162 inline static void timespec_add_us(struct timespec *t, long int us) {
163  t->tv_nsec += us;
164  if (t->tv_nsec >= 1000000000) {
165  t->tv_nsec = t->tv_nsec - 1000000000;
166  t->tv_sec += 1;

168 }
169 void timespec_sub_us(struct timespec *t, long int us) {
170  t->tv_nsec -= us;
171  if (t->tv_nsec < 0) {
172  t->tv_nsec = t->tv_nsec + 1000000000;
173  t->tv_sec -= 1;
174  }

177 void* nanoclock_timer_run_thread(rtdal_timer_t* obj) {
178  int s;
179 
180  assert(obj->period_function);
181 
182  obj->stop = 0;
183  if (obj->wait_futex) {
184  futex_wait(obj->wait_futex);
185  obj->next.tv_sec+=TIMER_FUTEX_GUARD_SEC;
186  obj->next.tv_nsec=0;
187  } else {
188  clock_gettime(CLOCK_MONOTONIC, &obj->next);
189  }
190  while(!obj->stop) {
191  timespec_add_us(&obj->next, obj->period);
192  clock_nanosleep(CLOCK_MONOTONIC, TIMER_ABSTIME,
193  &obj->next, NULL);
194  obj->period_function(obj->arg, &obj->next);
195  }
196 
197  s = 0;
198  pthread_exit(&s);
199  return NULL;
200 }
