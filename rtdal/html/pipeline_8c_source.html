53  hdebug("pipeid=%d, pid=%d, idx=%d, run=%d\n",pipe->id,proc->pid,idx,proc->runnable);

81  } else {
82  aerror_msg("Abnormal pid=%d termination but no callback was defined\n",
83  proc->pid);
84  }

107 inline static void pipeline_run_time_slot(pipeline_t *obj) {
108  int idx;
109  rtdal_process_t *run_proc;
110  hdebug("pipeid=%d, tslot=%d, nof_process=%d thread=%d\n",obj->id,obj->ts_counter,
111  obj->nof_processes, obj->thread);
112  obj->finished = 0;
113 
114  pipeline_run_thread_print_time(obj);
115 
116  run_proc = obj->first_process;
117  idx = 0;
118 
119  while(run_proc) {
120  hdebug("%d/%d: run=%d code=%d next=0x%x\n",idx,obj->nof_processes,run_proc->runnable,
121  run_proc->finish_code,run_proc->next);
122  if (idx > obj->nof_processes) {
123  aerror_msg("Fatal error. Corrupted pipeline-%d process list at process %d\n",
124  obj->id, idx);
125  kill(getpid(),SIGTERM);
126  pthread_exit(NULL);
127  }
128  pipeline_run_thread_check_status(obj,run_proc);
129  pipeline_run_thread_run_module(obj,run_proc, idx);
130  run_proc = run_proc->next;
131  idx++;
132  }
133  obj->ts_counter++;
134  obj->finished = 1;
135 }

165 void *pipeline_run_thread(void *self) {
166  assert(self);
167  pipeline_t *obj = (pipeline_t*) self;
168  assert(obj->id>=0);
169 
170  hdebug("pipeid=%d waiting\n",obj->id);
171 
172  obj->stop = 0;
173  pipeline_sync_thread_waits(obj->id);
174  hdebug("pipeid=%d start\n",obj->id);
175  while(!obj->stop) {
176  pipeline_run_time_slot(obj);
177  pipeline_sync_thread_waits(obj->id);
178  }
179  hdebug("pipeid=%d exiting\n",obj->id);
180  return NULL;

236 int pipeline_add(pipeline_t *obj, rtdal_process_t *process) {
237  hdebug("pipeid=%d, nof_process=%d, pid=%d, exec_pos=%d\n",obj->id,obj->nof_processes,
238  process->pid,process->attributes.exec_position);
239  RTDAL_ASSERT_PARAM(obj);
240  RTDAL_ASSERT_PARAM(process);
241  int exec_pos, i;
242  rtdal_process_t *p = NULL;
243 
244  exec_pos = process->attributes.exec_position;
245 
246  /* head because empty list */
247  if (!obj->first_process) {
248  hdebug("pipeid=%d add pid=%d to head\n", obj->id, process->pid);
249  obj->first_process = process;
250  process->next = NULL;
251  goto end;
252  }
253  /* head because first exec position */
254  if (exec_pos < obj->first_process->attributes.exec_position) {
255  hdebug("pipeid=%d add pid=%d to head\n", obj->id, process->pid);
256  process->next = obj->first_process;
257  obj->first_process = process;
258  goto end;
259  }
260  /* middle */
261  i=0;
262  p = obj->first_process;
263  while(p->next && exec_pos < p->next->attributes.exec_position) {
264  p=p->next;
265  i++;
266  }
267  process->next = p->next;
268  p->next = process;
269 
270  hdebug("pipeid=%d, add pid=%d to pos=%d\n", obj->id, process->pid,i);
271 
272 end:
273  obj->nof_processes++;
274  /* assign pipeline to object */
275  process->pipeline = obj;
276 
277  return i;
278 }

286 int pipeline_remove(pipeline_t *obj, rtdal_process_t *proc) {
287  hdebug("pipeid=%d, nof_process=%d, pid=%d, pid_pos=%d\n",obj->id,obj->nof_processes,
288  proc->pid,proc->attributes.exec_position);
289  RTDAL_ASSERT_PARAM(obj);
290  RTDAL_ASSERT_PARAM(proc);
291 
292  rtdal_process_t *cur, *prev;
293 
294  prev = NULL;
295  cur = obj->first_process;
296  while(cur != proc && cur) {
297  hdebug("pipeid=%d, prev=0x%x, cur=0x%x\n", obj->id,
298  prev,cur);
299  prev = cur;
300  cur = cur->next;
301  }
302  if (!cur) {
303  RTDAL_SETERROR(RTDAL_ERROR_NOTFOUND);
304  return -1;
305  }
306  if (prev) {
307  hdebug("pipeid=%d remove middle/end\n",obj->id);
308  prev->next = cur->next;
309  } else {
310  hdebug("pipeid=%d remove first\n",obj->id);
311  obj->first_process = cur->next;
312  }
313 
314  obj->nof_processes--;
315  proc->next = NULL;
316 
317  return 0;
318 }
