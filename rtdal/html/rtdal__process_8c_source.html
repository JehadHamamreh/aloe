35 int rtdal_process_launch(rtdal_process_t *obj) {
36  hdebug("path=%s\n",obj->attributes.binary_path);
37  RTDAL_ASSERT_PARAM(obj);
38  char *error;
39 
40  snprintf(tmp,LSTR_LEN,"/usr/local/lib/%s",obj->attributes.binary_path);
41  obj->dl_handle = dlopen(tmp, RTLD_NOW);
42 
43  if (!obj->dl_handle) {
44  RTDAL_DLERROR(dlerror());
45  return -1;
46  }
47  dlerror();
48 
49  *(void**) (&obj->run_point) = dlsym(obj->dl_handle, "_run_cycle");
50  if ((error = dlerror()) != NULL) {
51  RTDAL_DLERROR(error);
52  return -1;
53  }
54 
55  return 0;
56 }
