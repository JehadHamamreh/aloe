125 int rtdal_task_new_thread(pthread_t *thread, void *(*fnc)(void*), void *arg,
126  enum rtdal_task_type type, int prio, int cpu, int unblock_signals) {
127  hdebug("thread=0x%x,fnc=0x%x,arg=0x%x,type=%s,prio=%d,cpu=%d\n",thread,fnc,arg,
128  type==DETACHABLE?"detach":"join",prio,cpu);
129  RTDAL_ASSERT_PARAM(thread);
130  RTDAL_ASSERT_PARAM(fnc);
131  RTDAL_ASSERT_PARAM(prio>=0);
132  RTDAL_ASSERT_PARAM(cpu>=-1);
133 
134  pthread_attr_t attr;
135  int s,ret;
136  struct sched_param param;
137  cpu_set_t cpuset;
138  int i;
139 
140  ret = -1;
141 
142  s = pthread_attr_init(&attr);
143  if (s) {
144  RTDAL_POSERROR(s, "pthread_attr_init");
145  goto destroy_attr;
146  }
147 
148  s = pthread_attr_setinheritsched(&attr,PTHREAD_EXPLICIT_SCHED);
149  if (s) {
150  RTDAL_POSERROR(s,"pthread_attr_setinheritsched");
151  goto destroy_attr;
152  }
153 
154  s = pthread_attr_setdetachstate(&attr,
155  (type == DETACHABLE)?PTHREAD_CREATE_DETACHED:PTHREAD_CREATE_JOINABLE);
156  if (s) {
157  RTDAL_POSERROR(s, "pthread_attr_setdetachstate");
158  goto destroy_attr;
159  }
160 
161  if (cpu != -1) {
162  CPU_ZERO(&cpuset);
163  CPU_SET((size_t) cpu,&cpuset);
164  } else {
165  CPU_ZERO(&cpuset);
166  for (i=0;i<nof_cores;i++) {
167  CPU_SET((size_t) core_mapping[i],&cpuset);
168  }
169  }
170  s = pthread_attr_setaffinity_np(&attr, sizeof(cpu_set_t), &cpuset);
171  if (s) {
172  RTDAL_POSERROR(s, "pthread_attr_setaffinity_np");
173  goto destroy_attr;
174  }
175 
176  if (!getuid()) {
177  s = pthread_attr_setschedpolicy(&attr, (prio==TASK_DEFAULT_PRIORITY)?SCHED_RR:SCHED_FIFO);
178  if (s) {
179  if (s == EPERM) {
180  awarn("Not enough privileges to set task scheduling\n");
181  } else {
182  RTDAL_POSERROR(s,"pthread_attr_setschedpolicy");
183  goto destroy_attr;
184  }
185  }
186  }
187 
188  if (!getuid()) {
189  param.sched_priority = prio;
190  s = pthread_attr_setschedparam(&attr, &param);
191  if (s) {
192  if (s == EPERM) {
193  awarn("Not enough privileges to set task priority\n");
194  } else {
195  printf("prio=%d\n",prio);
196  RTDAL_POSERROR(s,"pthread_attr_setschedparam");
197  goto destroy_attr;
198  }
199  }
200  }
201 
202  hdebug("creating thread prio %d cpuid %d\n",prio,cpu);
203 
204  s = pthread_create(thread, &attr, fnc, arg);
205  if (s) {
206  if (s == EPERM) {
207  awarn("Not enough privileges to set kernel thread priority\n");
208  } else {
209  RTDAL_POSERROR(s, "pthread_create");
210  goto destroy_attr;
211  }
212  }
213 
214  ret=0;
215 
216 destroy_attr:
217  pthread_attr_destroy(&attr);
218  return ret;
219 }
