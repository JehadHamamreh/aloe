34 int rtdal_itfspscq_init(rtdal_itfspscq_t *itf) {
35  RTDAL_ASSERT_PARAM(itf);
36  int i;
37  char *x;
38 
39  itf->parent.is_external = 0;
40  itf->parent.delay = 1;
41  itf->read = 0;
42  itf->write = 0;
43  itf->data = malloc(itf->max_msg*itf->max_msg_sz);
44  itf->packets = calloc(itf->max_msg,sizeof(r_pkt_t));
45  if (!itf->data || !itf->packets) {
46  return -1;

48  for (i=0;i<itf->max_msg;i++) {
49  itf->packets[i].data = &itf->data[i*itf->max_msg_sz];
50  itf->packets[i].valid = 0;
51  }
52 
53  return 0;
54 }

83  cast(obj,itf);

86  *ptr = NULL;

101 int rtdal_itfspscq_push(r_itf_t obj, int len) {
102  cast(obj,itf);
103  if (spscq_is_full(itf)) {
104  RTDAL_SETERROR(RTDAL_ERROR_NOSPACE);
105  return 0;
106  }
107  itf->packets[itf->write].tstamp = rtdal_time_slot()+itf->parent.delay;
108  itf->packets[itf->write].len = len;
109  itf->packets[itf->write].valid = 1;
110  itf->write += (itf->write+1 >= itf->max_msg) ? (1-itf->max_msg) : 1;
111 
112  qdebug("write=%d/%d, len=%d\n",itf->write,itf->max_msg,len);
113 
114  return 1;
115 }

117 int rtdal_itfspscq_pop(r_itf_t obj, void **ptr, int *len) {
118  cast(obj,itf);
119  RTDAL_ASSERT_PARAM(ptr);
120  RTDAL_ASSERT_PARAM(len);
121 
122  *ptr = NULL;
123  *len = 0;
124 
125  int tstamp = rtdal_time_slot();
126  if (spscq_is_empty(itf)) {
127  qdebug("[empty] read=%d, tstamp=%d\n",itf->read,itf->packets[itf->read].tstamp);
128  return 0;
129  }
130  if (itf->packets[itf->read].tstamp > tstamp) {
131  qdebug("[delay] read=%d, tstamp=%d\n",itf->read,itf->packets[itf->read].tstamp);
132  return 0;
133  }
134  if (itf->packets[itf->read].tstamp < tstamp) {
135  qdebug("[late] read=%d tstamp=%d ara=%d\n",itf->read,itf->packets[itf->read].tstamp,tstamp);
136  }
137 
138  qdebug("[ok] read=%d, tstamp=%d (now=%d)\n",itf->read,itf->packets[itf->read].tstamp,tstamp);
139  *ptr = itf->packets[itf->read].data;
140  *len = itf->packets[itf->read].len;
141 
142  return 1;
143 }

186 int rtdal_itfspscq_send(r_itf_t obj, void* buffer, int len) {
187  cast(obj,itf);
188  RTDAL_ASSERT_PARAM(buffer);
189  RTDAL_ASSERT_PARAM(len>=0);
190 
191  int n;
192  void *ptr;
193 
194  if (len > itf->max_msg_sz) {
195  RTDAL_SETERROR(RTDAL_ERROR_LARGE);
196  return -1;
197  }
198 
199  hdebug("requesting pkt for 0x%x\n",obj);
200  if ((n = rtdal_itfspscq_request(obj, &ptr)) != 1) {
201  return n;
202  }
203 
204  memcpy(ptr, buffer, (size_t) len);
205 
206  hdebug("put pkt for 0x%x pkt 0x%x\n",obj,ptr);
207  return rtdal_itfspscq_push(obj,len);
208 }

210 int rtdal_itfspscq_recv(r_itf_t obj, void* buffer, int len) {
211  cast(obj,itf);
212 
213  RTDAL_ASSERT_PARAM(buffer);
214  RTDAL_ASSERT_PARAM(len>=0);
215 
216  int n, plen;
217  void *ptr;
218 
219  if ((n = rtdal_itfspscq_pop(obj, &ptr, &plen)) != 1) {
220  return n;
221  }
222  if (plen > len) {
223  plen = len;
224  }
225 
226  hdebug("obj=0x%x, rcv pkt=0x%x\n",obj,ptr);
227  memcpy(buffer, ptr, (size_t) plen);
228 
229  if ((n = rtdal_itfspscq_release(obj)) == 1) {
230  return n;
231  }
232 
233  hdebug("release pkt 0x%x\n",ptr);
234 
235  return plen;
236 }
