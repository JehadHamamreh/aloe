75 void rtdal_timeslot_set(int ts_base_multiply) {
76  switch(rtdal.machine.clock_source) {
77  case SINGLE_TIMER:
78  kernel_timer.multiple = ts_base_multiply;
79  break;
80  case MULTI_TIMER:
81  for (int i=0;i<rtdal.machine.nof_cores;i++) {
82  rtdal.pipelines[i].mytimer.multiple = ts_base_multiply;
83  }
84  break;
85  default:
86  aerror("Not implemented\n");
87  break;
88  }

91 inline static int kernel_initialize_setup_clock() {
92  struct timespec start_time;
93 #ifdef HAVE_UHD
94  struct sched_param param;
95 #endif
96 
97  /* access to kernel sync function is not allowed */
98  rtdal.machine.slave_sync_kernel = NULL;
99 
100  switch(rtdal.machine.clock_source) {
101  case SINGLE_TIMER:
102  kernel_timer.period_function = kernel_cycle;
103  kernel_timer.period = rtdal.machine.ts_len_us*1000;
104  kernel_timer.arg = NULL;
105  kernel_timer.multiple = 1;
106 #ifdef __XENO__
107  kernel_timer.mode = XENOMAI;
108 #else
109  kernel_timer.mode = NANOSLEEP;
110 #endif
111  kernel_timer.wait_futex = NULL;
112  kernel_timer.thread = &single_timer_thread;
113  hdebug("creating single_timer_thread period %d\n",(int) kernel_timer.period);
114  if (rtdal_task_new_thread(&single_timer_thread, timer_run_thread,
115  &kernel_timer, DETACHABLE,
116  rtdal.machine.kernel_prio, 0,0)) {
117  rtdal_perror("rtdal_task_new_thread");
118  return -1;
119  }
120  break;
121  case MULTI_TIMER:
122  usleep(100000);
123  printf("Starting clocks in %d sec...\n", TIMER_FUTEX_GUARD_SEC);
124  clock_gettime(CLOCK_REALTIME, &start_time);
125  for (int i=0;i<rtdal.machine.nof_cores;i++) {
126  rtdal.pipelines[i].mytimer.next = start_time;
127  rtdal.pipelines[i].mytimer.multiple = 1;
128  }
129  futex_wake(&multi_timer_futex);
130  break;
131  case DAC:
132 #ifdef HAVE_UHD
133  param.sched_priority = rtdal.machine.kernel_prio-2;
134  pthread_setschedparam(pthread_self(),SCHED_FIFO,&param);
135  uhd_init(&dac_cfg, &rtdal.machine.ts_len_us,dac_cycle);
136  param.sched_priority = rtdal.machine.kernel_prio;
137  pthread_setschedparam(pthread_self(),SCHED_FIFO,&param);
138 #endif
139  break;
140  case SYNC_SLAVE:
141  /* enable access to kernel_cycle function */
142  rtdal.machine.slave_sync_kernel = kernel_cycle;
143  break;
144  default:
145  aerror_msg("Unknown clock source %d\n", rtdal.machine.clock_source);
146  return -1;
147  }
148  return 0;
149 }

151 int kernel_initialize_create_pipeline(pipeline_t *obj, int *wait_futex) {
152  void *(*tmp_thread_fnc)(void*);
153  void *tmp_thread_arg;
154  int prio;
155 
156  hdebug("pipeline_id=%d\n",obj->id);
157  if (rtdal.machine.clock_source == MULTI_TIMER) {
158  obj->mytimer.period_function =
159  pipeline_run_from_timer;
160  obj->mytimer.period =
161  rtdal.machine.ts_len_us*1000;
162  obj->mytimer.arg = obj;
163  obj->mytimer.wait_futex = wait_futex;
164 #ifdef __XENO__
165  obj->mytimer.mode = XENOMAI;
166 #else
167  obj->mytimer.mode = NANOSLEEP;
168 #endif
169  obj->mytimer.thread =
170  &obj->thread;
171  tmp_thread_fnc = timer_run_thread;
172  tmp_thread_arg = &obj->mytimer;
173  prio = rtdal.machine.kernel_prio;
174  } else {
175  tmp_thread_fnc = pipeline_run_thread;
176  tmp_thread_arg = obj;
177  prio = rtdal.machine.kernel_prio-1;
178  }
179 
180  if (rtdal_task_new_thread(&obj->thread,
181  tmp_thread_fnc, tmp_thread_arg, DETACHABLE,
182  prio, core_mapping[obj->id],0)) {
183  rtdal_perror();
184  return -1;
185  }
186  return 0;
187 }

211 int kernel_initialize_set_kernel_priority() {
212  struct sched_param param;
213  cpu_set_t cpuset;
214 
215  param.sched_priority = rtdal.machine.kernel_prio;
216  hdebug("kernel_prio=%d\n",rtdal.machine.kernel_prio);
217 
218  if (sched_setscheduler(0, SCHED_FIFO, &param)) {
219  if (errno == EPERM) {
220  awarn("Not enough privileges to set kernel thread priority\n");
221  } else {
222  poserror(errno, "sched_setscheduler");
223  return -1;
224  }
225  }
226 
227  CPU_ZERO(&cpuset);
228  CPU_SET(0, &cpuset);
229 
230  if (sched_setaffinity(0, sizeof(cpu_set_t), &cpuset)) {
231  poserror(errno, "sched_setaffinity");
232  return -1;
233  }
234 
235  return 0;
236 }

248 static int kernel_initialize(void) {
249 
250  /* Initialize rtdal_base library */
251  rtdal.machine.clock_source = clock_source;
252  rtdal.machine.ts_len_us = timeslot_us;
253  rtdal.machine.kernel_prio = KERNEL_RT_PRIO;
254  rtdal.machine.rt_fault_opts = RT_FAULT_OPTS_HARD;
255  rtdal_initialize_node(&rtdal, NULL, NULL);
256 
257  pthread_mutex_init(&rtdal.mutex,NULL);
258 
259 
260  /* Set self priority to rtdal.machine.kernel_prio */
261 #ifdef KERNEL_SIGWAIT_RT_PRIO
262  if (kernel_initialize_set_kernel_priority()) {
263  return -1;
264  }
265 #endif
266 
267  /* Setup signals */
268  if (kernel_initialize_setup_signals()) {
269  return -1;
270  }
271 
272  /* create pipelines */
273  if (kernel_initialize_create_pipelines()) {
274  return -1;
275  }
276 
277  /* setup clock */
278  if (kernel_initialize_setup_clock()) {
279  return -1;
280  }
281 
282 
283  return 0;
284 }

288 static void check_threads() {
289  int i;
290  hdebug("nof_cores=%d, timer=%d\n",rtdal.machine.nof_cores,single_timer_thread);
291  if (single_timer_thread) {
292  if (!pthread_kill(single_timer_thread,0)) {
293  aerror("kernel timer still running, killing\n");
294  pthread_kill(single_timer_thread, TASK_TERMINATION_SIGNAL);
295  }
296  }
297  for (i=0;i<rtdal.machine.nof_cores;i++) {
298  hdebug("thread_%d=%d\n",i,rtdal.pipelines[i].thread);
299  if (rtdal.pipelines[i].thread) {
300  if (!pthread_kill(rtdal.pipelines[i].thread,0)) {
301  aerror_msg("pipeline thread %d still running, killing\n",i);
302  pthread_kill(rtdal.pipelines[i].thread, TASK_TERMINATION_SIGNAL);
303  }
304  } else {
305  if (rtdal.pipelines[i].waiting) {
306  hdebug("setting wait to zero\n",0);
307  rtdal.pipelines[i].waiting = 0;
308  usleep(20000);
309  hdebug("done",0);
310  }
311  }
312  }
313  hdebug("i=%d\n",i);
314 }

343 void kernel_exit() {
344  write_debug_trace();
345  hdebug("tslot=%d\n",rtdal_time_slot());
346  sigwait_stops = 1;
347  sem_post(&dac_sem);
348  kernel_timer.stop = 1;
349  for (int i=0;i<rtdal.machine.nof_cores;i++) {
350  if (rtdal.machine.clock_source == MULTI_TIMER) {
351  rtdal.pipelines[i].mytimer.stop = 1;
352  } else {
353  rtdal.pipelines[i].stop = 1;
354  }
355  }
356 #ifdef HAVE_UHD
357  if (using_uhd) {
358  uhd_close();
359  }
360 #endif
361  usleep(100000);
362  check_threads();
363 }

365 int main(int argc, char **argv) {
366 
367  mlockall(MCL_CURRENT | MCL_FUTURE);
368 
369  print_license();
370  if (getuid()) {
371  printf("Run as root to run in real-time mode\n\n");
372  }
373 
374  if (argc!=3) {
375  printf("Usage: %s path_to_waveform_model config_file\n",argv[0]);
376  return -1;
377  }
378 
379  if (parse_config(argv[2])) {
380  aerror_msg("Error parsing file config %s\n",argv[2]);
381  exit(0);
382  }
383 
384 #ifdef DEBUG_TRACE
385  open_debug_trace();
386  atexit(write_debug_trace);
387 #endif
388 
389 #ifdef __XENO__
390  pthread_set_mode_np(0, PTHREAD_WARNSW);
391 #endif
392  kernel_pid = getpid();
393 
394  rtdal.machine.nof_cores = nof_cores;
395 
396  if (using_uhd) {
397 #ifdef HAVE_UHD
398  struct sched_param param;
399  sem_init(&dac_sem, 0, 0);
400  param.sched_priority = rtdal.machine.kernel_prio-3;
401  pthread_setschedparam(pthread_self(),SCHED_FIFO,&param);
402  uhd_init(&dac_cfg, &rtdal.machine.ts_len_us,dac_cycle);
403  param.sched_priority = rtdal.machine.kernel_prio;
404  pthread_setschedparam(pthread_self(),SCHED_FIFO,&param);
405 #endif
406  }
407 
408  printf("Time slot:\t%d us\nPlatform:\t%d cores\nTimer:\t\t%s\n\n", (int) timeslot_us,
409  rtdal.machine.nof_cores,(clock_source==MULTI_TIMER)?"Multi":"Single");
410 
411 
412  /* initialize kernel */
413  if (kernel_initialize()) {
414  aerror("Initiating kernel\n");
415  goto clean_and_exit;
416  }
417 
418  if (rtdal_task_new(NULL,_run_main,argv[1])) {
419  rtdal_perror("rtdal_task_new");
420  goto clean_and_exit;
421  }
422 
423  /* the main thread runs the sigwait loop */
424  sigwait_loop();
425 
426  printf("exiting\n");
427 
428 clean_and_exit:
429  exit(0);
430 }
