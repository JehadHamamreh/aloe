44 int rtdal_process_launch(rtdal_process_t *obj) {
45  hdebug("path=%s\n",obj->attributes.binary_path);
46  RTDAL_ASSERT_PARAM(obj);
47  char *error;
48 
49  char *name = strstr(obj->attributes.binary_path,"/");
50  if (!name) {
51  name = obj->attributes.binary_path;
52  } else {
53  name++;
54  }
55 
56  snprintf(tmp,LSTR_LEN,"/tmp/am_%s_%d_%d.so",name,
57  obj->pid,kernel_pid);
58  snprintf(tmp2,LSTR_LEN,"cp %s/%s %s",libs_path,obj->attributes.binary_path,tmp);
59  if (system(tmp2) == -1) {
60  aerror("Error removing file\n");
61  return -1;
62  }
63 
64  obj->dl_handle = dlopen(tmp,RTLD_NOW);
65  if (!obj->dl_handle) {
66  RTDAL_DLERROR(dlerror());
67  return -1;
68  }
69 
70  dlerror();
71 
72  *(void**) (&obj->run_point) = dlsym(obj->dl_handle, "_run_cycle");
73  if ((error = dlerror()) != NULL) {
74  RTDAL_DLERROR(error);
75  return -1;
76  }
77 
78  return 0;
79 }
