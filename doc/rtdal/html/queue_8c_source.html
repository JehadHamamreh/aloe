25 void queue_init(queue_t *q, int pkt_sz, int nof_pkts) {
26  int i;
27 
28  q->read = 0;
29  q->write = 0;
30  q->size = nof_pkts;
31  q->data = malloc(nof_pkts*pkt_sz);
32  q->packets = malloc(nof_pkts*sizeof(h_pkt_t));

37 }

66 int queue_push(queue_t *q) {
67  qdebug("write=%d, tstamp=%d\n",q->write,q->packets[q->write].tstamp);
68  q->packets[q->write].valid = 1;
69  q->write += (q->write+1 >= q->size) ? (1-q->size) : 1;

73 h_pkt_t *queue_pop(queue_t *q, int tstamp) {
74 
75 
76  if (queue_is_empty(q)) {
77  qdebug("[empty] read=%d, tstamp=%d\n",q->read,q->packets[q->read].tstamp);
78  return NULL;
79  }
80  if (tstamp) {
81  if (q->packets[q->read].tstamp > tstamp) {
82  qdebug("[delay] read=%d, tstamp=%d\n",q->read,q->packets[q->read].tstamp);
83  return NULL;
84  }
85  if (q->packets[q->read].tstamp < tstamp) {
86  qdebug("[late] read=%d tstamp=%d ara=%d\n",q->read,q->packets[q->read].tstamp,tstamp);
87  }
88  }
89  qdebug("[ok] read=%d, tstamp=%d (now=%d)\n",q->read,q->packets[q->read].tstamp,tstamp);
90  return &q->packets[q->read];
91 }
