148 inline static int kernel_initialize_setup_clock() {
149  struct timespec start_time;
150 
151  /* access to kernel sync function is not allowed */
152  rtdal.machine.slave_sync_kernel = NULL;
153 
154  switch(rtdal.machine.clock_source) {
155  case SINGLE_TIMER:
156  kernel_timer.period_function = kernel_cycle;
157  kernel_timer.period = rtdal.machine.ts_len_us*1000;
158  kernel_timer.arg = NULL;
159  kernel_timer.mode = NANOSLEEP;
160  kernel_timer.wait_futex = NULL;
161  kernel_timer.thread = &single_timer_thread;
162  hdebug("creating single_timer_thread period %d\n",(int) kernel_timer.period);
163  if (rtdal_task_new_thread(&single_timer_thread, timer_run_thread,
164  &kernel_timer, DETACHABLE,
165  rtdal.machine.kernel_prio-1, 0,0)) {
166  rtdal_perror("rtdal_task_new_thread");
167  return -1;
168  }
169  break;
170  case MULTI_TIMER:
171  usleep(100000);
172  printf("Starting clocks in %d sec...\n", TIMER_FUTEX_GUARD_SEC);
173  clock_gettime(CLOCK_MONOTONIC, &start_time);
174  for (int i=0;i<rtdal.machine.nof_cores;i++) {
175  rtdal.pipelines[i].mytimer.next = start_time;
176  }
177  futex_wake(&multi_timer_futex);
178  break;
179  case DAC:
180  case SYNC_SLAVE:
181  /* enable access to kernel_cycle function */
182  rtdal.machine.slave_sync_kernel = kernel_cycle;
183  break;
184  default:
185  aerror_msg("Unknown clock source %d\n", rtdal.machine.clock_source);
186  return -1;
187  }
188  return 0;
189 }

248 inline static int kernel_initialize_setup_signals() {
249  int s;
250  sigset_t set;
251  int i;
252  struct sigaction action;
253 
254  action.sa_sigaction = thread_signal_handler;
255  action.sa_flags = SA_SIGINFO;
256 
257  sigfillset(&set);
258  /* do not block thread-specific signals. Add a handler to them */
259  for (i=0;i<N_THREAD_SPECIFIC_SIGNALS;i++) {
260  sigdelset(&set, thread_specific_signals[i]);
261  if (sigaction(thread_specific_signals[i], &action, NULL)) {
262  poserror(errno, "sigaction");

265  }

267  if (s != 0) {

354 static int sigwait_loop_process_thread_signal(siginfo_t *info) {
355  int thread_id, signum;
356  strdef(tmp_msg);
357 
358  signum = info->si_value.sival_int & 0xffff;
359  thread_id = info->si_value.sival_int >> 16;
360  hdebug("signum=%d, thread_id=%d\n",signum,thread_id);
361  if (signum < N_THREAD_SPECIFIC_SIGNALS) {
362  sprintf(tmp_msg, "Got signal num %d from ",
363  thread_specific_signals[signum]);
364  } else {
365  sprintf(tmp_msg, "Got unknown signal from ");
366  }
367 
368  /* now try to restore the pipeline, if the thread was a pipeline */
369  if (thread_id > -1) {
370  if (strlen(tmp_msg)>1) {
371  sprintf(&tmp_msg[strlen(tmp_msg)-1],
372  "pipeline thread %d process %d\n",
373  thread_id,
374  rtdal.pipelines[thread_id].running_process_idx);
375  }
376  if (pipeline_recover_thread(
377  &rtdal.pipelines[thread_id])) {
378  aerror("recovering pipeline thread\n");
379  }
380 
381  } else if (info->si_value.sival_int == -1) {
382  strcat(tmp_msg, "the kernel thread\n");
383  kernel_timer_recover_thread();
384  } else {
385  strcat(tmp_msg, "an unkown thread\n");
386  }
387  return 1;
388 }

402 static void sigwait_loop(void) {
403 
404  int signum;
405  sigset_t set;
406  siginfo_t info;
407 
408  sigfillset(&set);
409  sigdelset(&set,TASK_TERMINATION_SIGNAL);
410  while(!sigwait_stops) {
411  do {
412  signum = sigwaitinfo(&set, &info);
413  } while (signum == -1 && errno == EINTR);
414  if (signum == -1) {
415  poserror(errno, "sigwaitinfo");
416  goto out;
417  }
418  hdebug("detected signal %d\n",signum);
419  if (signum == KERNEL_SIG_THREAD_SPECIFIC) {
420  sigwait_loop_process_thread_signal(&info);
421  } else if (signum == SIGINT) {
422  printf("Caught SIGINT, exiting\n");
423  fflush(stdout);
424  goto out;
425  } else {
426  printf("Got signal %d, exiting\n", signum);
427  fflush(stdout);
428  goto out;
429  }
430  }
431 
432 out:
433  go_out();
434 }

444 static void thread_signal_handler(int signum, siginfo_t *info, void *ctx) {
445  union sigval value;
446  int thread_id;
447  int i;
448 
449 #ifdef PRINT_BT_ON_SIGSEGV
450  void *pnt = NULL;
451  ucontext_t *context = (ucontext_t*) ctx;
452  char **msg;
453 
454  pnt = (void*) context->uc_mcontext.gregs[REG_EIP];
455  msg = backtrace_symbols(&pnt, 1);
456  strcat(sigmsg, msg[0]);
457  free(msg);
458 #endif
459 
460  signal_received++;
461 
462  printf("[ts=%d] signal %d received\n",rtdal_time_slot(),signum);
463 
464  /* try to find the thread that caused the signal */
465 
470  pthread_t thisthread = pthread_self();
471 
472  /* if signum is SIGUSR2, its a task termination signal, just exit */
473  if (signum == TASK_TERMINATION_SIGNAL) {
474  hdebug("sigusr2 signal. thread=%d\n",thisthread);
475  goto cancel_and_exit;
476  }
477 
478  /* is it a pipeline thread? */
479  for (i=0;i<rtdal.machine.nof_cores;i++) {
480  if (thisthread == rtdal.pipelines[i].thread) {
481  break;
482  }
483  }
484  if (i < rtdal.machine.nof_cores) {
485  hdebug("pipeline_idx=%d\n",i);
486  thread_id = i;
487 
488  /* set the thread to 0 because is terminating */
489  rtdal.pipelines[thread_id].thread = 0;
490  } else {
491  /* it is not, may be it is the kernel timer */
492  if (thisthread == single_timer_thread) {
493  hdebug("timer thread=%d\n",thisthread);
494  thread_id = -1;
495  } else {
496  /* @TODO: check if it is a status or init thread of any module */
497 
498  hdebug("other thread=%d\n",thisthread);
499  goto cancel_and_exit;
500  }
501  }
502 
503  /* Now send a signal to the kernel */
504  for (i=0;i<N_THREAD_SPECIFIC_SIGNALS;i++) {
505  if (thread_specific_signals[i] == signum)
506  break;
507  }
508  hdebug("signal=%d, thread=%d\n",i,thread_id);
509  value.sival_int = thread_id<<16 | i;
510  if (sigqueue(kernel_pid,
511  KERNEL_SIG_THREAD_SPECIFIC,
512  value)) {
513  poserror(errno, "sigspscq");
514  }
515 
516 cancel_and_exit:
517  pthread_exit(NULL);
518 }
